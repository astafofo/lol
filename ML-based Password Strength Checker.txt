import math

# Simple password entropy calculation: entropy = length * log2(charset_size)
def password_entropy(pwd):
    charset_size = 26 + 26 + 10 + 32  # lowercase+uppercase+digits+symbols
    return len(pwd) * math.log2(charset_size)

def classify_password(pwd):
    e = password_entropy(pwd)
    if e < 50:
        return "Weak"
    elif e < 70:
        return "Medium"
    else:
        return "Strong"

passwords = ["12345", "Abc123!", "A$tr0ngP@ssword2025"]
for pwd in passwords:
    print(f"{pwd}: {classify_password(pwd)}")
