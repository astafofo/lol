import numpy as np
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import LSTM, Dense

# Sample time-series network traffic: [timesteps, features]
X = np.random.rand(2,5,3)  # 2 samples, 5 timesteps, 3 features
y = np.array([0,1])

model = Sequential([LSTM(8, input_shape=(5,3)), Dense(1, activation='sigmoid')])
model.compile(optimizer='adam', loss='binary_crossentropy', metrics=['accuracy'])
model.fit(X, y, epochs=2, verbose=0)
print("LSTM trained on sample network traffic.")
